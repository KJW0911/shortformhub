import sqlite3
import feedparser
from bs4 import BeautifulSoup

# --- 1단계: 데이터베이스 및 테이블 초기화 함수 ---
def init_db():
    conn = sqlite3.connect("reddit_data.db")
    cursor = conn.cursor()
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS reddit_posts (
            post_id TEXT PRIMARY KEY,
            category TEXT,
            title TEXT,
            author TEXT,
            link TEXT,
            content TEXT,
            published_at TEXT,
            collected_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    """)
    conn.commit()
    return conn

# --- 2단계: HTML 태그를 제거하는 텍스트 정제 함수 ---
def clean_html(html_content):
    if not html_content:
        return ""
    return BeautifulSoup(html_content, "html.parser").get_text()

# --- 3단계: 뉴스 및 인기 카테고리를 순회하며 수집하는 로직 ---
def collect_and_save_reddit():
    conn = init_db()
    cursor = conn.cursor()
    
    rss_feeds = {
        "뉴스(News)": "https://www.reddit.com/r/news/.rss?sort=new",
        "인기(Popular)": "https://www.reddit.com/r/popular/.rss"
    }
    
    total_new_posts = 0
    
    for category_name, rss_url in rss_feeds.items():
        print(f"\n🚀 [{category_name}] 카테고리 수집 시작...")
        
        feed = feedparser.parse(rss_url, agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36')
        category_posts_count = 0
        
        for entry in feed.entries:
            post_id = entry.id
            title = entry.title
            author = entry.author
            link = entry.link
            published_at = entry.updated if hasattr(entry, 'updated') else (entry.published if hasattr(entry, 'published') else "")
            content = clean_html(entry.summary) if hasattr(entry, 'summary') else ""
            
            try:
                cursor.execute("""
                    INSERT OR IGNORE INTO reddit_posts (post_id, category, title, author, link, content, published_at)
                    VALUES (?, ?, ?, ?, ?, ?, ?)
                """, (post_id, category_name, title, author, link, content, published_at))
                
                if cursor.rowcount > 0:
                    category_posts_count += 1
                    total_new_posts += 1
                    print(f"   ✨ [새 글 발견] {title[:25]}...")
                    
            except sqlite3.Error as e:
                print(f"   DB 오류 발생: {e}")
                
        print(f"✅ [{category_name}] 완료! 새로 추가된 글: {category_posts_count}개")
            
    conn.commit()
    conn.close()
    
    print(f"\n🎉 전체 수집 마감! 오늘 새로 추가된 총 게시글: {total_new_posts}개")

if __name__ == "__main__":
    collect_and_save_reddit()
